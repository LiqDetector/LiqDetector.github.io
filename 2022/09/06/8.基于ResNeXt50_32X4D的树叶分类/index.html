class LeavesData(Dataset):
  def __init__(self, csv_path, file_path, mode='train', valid_ratio=0.2, resize_height=256, resize_width=256):
    '''
    csv_path: csv文件路径
    file_path: 主文件路径
    mode: 训练模式还是测试模式
    valid_ratio: 划分训练集和验证集的比例
    resize_height: 调整图片

    1、调整图片尺寸
    2、读取csv文件到pd
    3、根据训练模式，加载图片路径、标签
    '''
  
    # 调整图片尺寸
    self.resize_height = resize_height
    self.resize_width = resize_width
    
    # 读取csv文件
    self.csv_path = csv_path
    self.file_path = file_path
    self.mode = mode
    # 利用pandas读取csv文件
    self.data_info = pd.read_csv(csv_path, header=None)

    self.data_len = len(self.data_info.index) - 1
    self.train_len = int(self.data_len * (1 - valid_ratio))

    if mode == 'train':
      self.train_image = np.asarray(self.data_info.iloc[1:self.train_len, 0])
      self.train_label = np.asarray(self.data_info.iloc[1:self.train_len, 1])
      self.image_arr = self.train_image
      self.label_arr = self.train_label
    elif mode == 'valid':
      self.valid_image = np.asarray(self.data_info.iloc[self.train_len+1:, 0])
      self.valid_label = np.asarray(self.data_info.iloc[self.train_len+1:, 1])
      self.image_arr = self.valid_image
      self.label_arr = self.valid_label
    else:
      self.test_image = np.asarray(self.data_info.iloc[1:, 0])
      self.image_arr = self.test_image

    self.real_len = len(self.image_arr)

    print('Finished reading the {} set of Leaves Dataset ({} samples found)'
              .format(mode, self.real_len))
  
  def __getitem__(self, index):
    '''
    1、根据index读取图片
    2、数据增强
    3、返回每一个index对应的图片和标签
    '''
    
    # 得到对应文件名
    single_image_name = self.image_arr[index]
    # 读取图像文件
    img_as_img = Image.open(self.file_path + single_image_name)
    # 只有训练集做数据增强
    if self.mode == 'train':
      transform = transforms.Compose([
          transforms.Resize((224,224)),
          transforms.RandomHorizontalFlip(p=0.5),
          transforms.ToTensor()
      ])
    else:
      transform = transforms.Compose([
          transforms.Resize((224,224)),
          transforms.ToTensor()
      ])
    img_as_img = transform(img_as_img)

    if self.mode == 'test':
      return img_as_img
    else:
      label = self.label_arr[index]
      number_label = class_to_num[label]
      # 返回每一个index对应的图片数据和对应的label
      return img_as_img, number_label

  def __len__(self):
    return self.real_len

# 加载预训练的模型
net = torchvision.models.resnext50_32x4d(weights=torchvision.models.ResNeXt50_32X4D_Weights.IMAGENET1K_V2)
net.fc = nn.Linear(net.fc.in_features, 176)

训练

from torch._C import device
learning_rate = 3e-4
weight_decay = 1e-3
num_epoch = 50

device = 'cuda'
net = net.to(device)
net.device = device
# 损失函数
criterion = nn.CrossEntropyLoss()
# 优化函数
optimizer = torch.optim.Adam(net.parameters(), lr=learning_rate, weight_decay=weight_decay)
# 训练轮数
n_epochs = num_epoch

best_acc = 0.0
for epoch in range(n_epochs):
  # 训练
  net.train()
  train_loss = []
  train_accs = []
  for imgs, labels in tqdm(train_loader):
    imgs = imgs.to(device)
    labels = labels.to(device)
    # 前向传播
    logits = net(imgs)
    # 计算损失值
    loss = criterion(logits, labels)
    # 清除上一步缓存的参数
    optimizer.zero_grad()
    # 反向传播计算梯度
    loss.backward()
    # 更新参数
    optimizer.step()
    # 求准确率
    acc = (logits.argmax(dim=-1) == labels).float().mean()
    # 记录
    train_loss.append(loss.item())
    train_accs.append(acc)
  
  train_loss = sum(train_loss) / len(train_loss)
  train_acc = sum(train_accs) / len(train_accs)

  print(f"[ Train | {epoch + 1:03d}/{n_epochs:03d} ] loss = {train_loss:.5f}, acc = {train_acc:.5f}")

  net.eval()
  valid_loss = []
  valid_accs = []
  for imgs,labels in tqdm(val_loader):
    imgs = imgs.to(device)
    labels = labels.to(device)
    # 停止梯度计算
    with torch.no_grad():
      logits = net(imgs)
    loss = criterion(logits, labels)
    # 验证集不需要优化器
    acc = (logits.argmax(dim=-1) == labels).float().mean()

    valid_loss.append(loss.item())
    valid_accs.append(acc)
  
  valid_los = sum(valid_loss) / len(valid_loss)
  valid_acc = sum(valid_accs) / len(valid_accs)

  print(f"[ Valid | {epoch + 1:03d}/{n_epochs:03d} ] loss = {valid_los:.5f}, acc = {valid_acc:.5f}")

  if valid_acc > best_acc:
    best_acc = valid_acc
    torch.save(net.state_dict(),'./pre_model.pth')
    print('Saving model with acc{:.3f}'.format(best_acc))
